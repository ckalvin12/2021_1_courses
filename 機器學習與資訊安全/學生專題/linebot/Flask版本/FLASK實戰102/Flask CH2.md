```
# 从flask框架中导入Flask类
from flask import Flask
# 传入__name__初始化一个Flask实例
app = Flask(__name__)
#这个路由将根URL映射到了hello_world函数上
@app.route('/')
def hello_world():
    return 'Hello World!'
if __name__ == '__main__':
    #指定默认主机为是127.0.0.1，port为8888
    app.run(debug=True,host='0.0.0.0', port=8888)

##### app.py
```
#encoding:utf-8
from flask import Flask
app = Flask(__name__)
@app.route('/')
def hello_world():
    return '这是url传参演示!'
@app.route('/user/<name>')
def list_name(name):
    return "接收到的名字为： %s" % name
@app.route('/news/<int:id>')
def list_news(id):
    return "接收到的id为： %s" % id
if __name__ == '__main__':
    app.run(debug=True)
